function fmtDate(ts) {
  const d = new Date(ts);
  return d.toLocaleDateString('en-IN',{day:'numeric',month:'short',year:'numeric'})+' '+d.toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'});
}
function relDate(ts) {
  const diff = Date.now()-ts;
  if(diff<60000)return'just now';
  if(diff<3600000)return Math.floor(diff/60000)+'m ago';
  if(diff<86400000)return Math.floor(diff/3600000)+'h ago';
  if(diff<604800000)return Math.floor(diff/86400000)+'d ago';
  return fmtDate(ts);
}

let recognition=null;
let isRecording=false;

function startVoice(){
  if(!currentNoteId)return showToast('error','Open a note first');
  const SpeechRecognition=window.SpeechRecognition||window.webkitSpeechRecognition;
  if(!SpeechRecognition)return showToast('error','🎙️ Speech not supported in this browser. Try Chrome.');
  if(isRecording){stopVoice();return;}
  recognition=new SpeechRecognition();
  recognition.continuous=true; recognition.interimResults=true; recognition.lang='en-US';
  let finalTranscript='';
  recognition.onstart=()=>{isRecording=true;document.getElementById('voiceIndicator').classList.add('show');showToast('info','🎙️ Listening… speak now');};
  recognition.onresult=e=>{
    let interim='';
    for(let i=e.resultIndex;i<e.results.length;i++){
      if(e.results[i].isFinal)finalTranscript+=e.results[i][0].transcript+' ';
      else interim+=e.results[i][0].transcript;
    }
  };
  recognition.onerror=e=>{showToast('error','🎙️ Error: '+e.error);stopVoice();};
  recognition.onend=()=>{
    if(finalTranscript.trim()){
      const ta=document.getElementById('editorBody');
      const pos=ta.selectionStart;
      const ins=(ta.value?'\n':'')+finalTranscript.trim();
      ta.value=ta.value.substring(0,pos)+ins+ta.value.substring(pos);
      ta.selectionStart=ta.selectionEnd=pos+ins.length;
      onEditorChange();
      showToast('success','🎙️ Voice transcribed!');
    }
    isRecording=false;
    document.getElementById('voiceIndicator').classList.remove('show');
  };
  recognition.start();
}
function stopVoice(){
  if(recognition)recognition.stop();
  isRecording=false;
  document.getElementById('voiceIndicator').classList.remove('show');
}

// IMAGE TO TEXT (OCR via Anthropic Vision API)
// ═══════════════════════════════════════════════════════
function handleImageOCR(e){
  if(!currentNoteId)return showToast('error','Open a note first');
  const file=e.target.files[0]; if(!file)return;
  const reader=new FileReader();
  reader.onload=async ev=>{
    const b64=ev.target.result.split(',')[1];
    const mtype=file.type||'image/jpeg';
    document.getElementById('ocrPreviewImg').src=ev.target.result;
    document.getElementById('imagePreviewArea').classList.add('show');
    document.getElementById('ocrResult').textContent='✦ Analyzing image…';
    ocrText='';
    try {
      const res=await fetch('https://api.anthropic.com/v1/messages',{
        method:'POST',
        headers:{'Content-Type':'application/json'},
        body:JSON.stringify({
          model:'claude-sonnet-4-20250514',
          max_tokens:1000,
          messages:[{role:'user',content:[
            {type:'image',source:{type:'base64',media_type:mtype,data:b64}},
            {type:'text',text:'Extract all visible text from this image. Return only the extracted text, preserving line breaks and structure. If no text, describe the image briefly.'}
          ]}]
        })
      });
      const data=await res.json();
      ocrText=data.content?.map(c=>c.text||'').join('\n').trim()||'No text found in image.';
      document.getElementById('ocrResult').textContent=ocrText;
      showToast('success','🖼️ Text extracted!');
    } catch(err){
      document.getElementById('ocrResult').textContent='Could not process image. Check connection.';
      showToast('error','OCR failed');
    }
  };
  reader.readAsDataURL(file);
  e.target.value='';
}
function insertOCRText(){
  if(!ocrText)return;
  const ta=document.getElementById('editorBody');
  const pos=ta.selectionStart;
  const ins=(ta.value?'\n\n':'')+ocrText;

  `;

function processFiles(files){
  if(!currentNoteId){showToast('error','Open a note first');return;}
  const n=notes.find(x=>x.id===currentNoteId); if(!n)return;
  if(!n.files)n.files=[];
  files.forEach(f=>{
    if(f.size>10*1024*1024){showToast('error',f.name+' too large (max 10MB)');return;}
    n.files.push({name:f.name,size:f.size,type:f.type,date:Date.now()});
    showToast('success','📎 '+f.name+' attached');
  });
  saveNotes(); renderAttachedFiles(n); renderNotesList();
}
function renderAttachedFiles(n){
  const c=document.getElementById('attachedFiles');
  if(!n.files||n.files.length===0){c.innerHTML='';return;}
  c.innerHTML=n.files.map((f,i)=>{
    const icon=f.type.includes('pdf')?'📄':f.type.includes('image')?'🖼️':'📃';
    const size=f.size>1024*1024?(f.size/1024/1024).toFixed(1)+'MB':(f.size/1024).toFixed(0)+'KB';
    return`<div class="file-chip">${icon} ${escHtml(f.name)} <span style="color:var(--text3)">${size}</span> <span class="remove" onclick="removeFile('${n.id}',${i})">✕</span></div>`;
  }).join('');
}
function removeFile(noteId,idx){
  const n=notes.find(x=>x.id===noteId); if(!n)return;
  n.files.splice(idx,1); saveNotes(); renderAttachedFiles(n); renderNotesList();
}

function updateStats(){
  const active=notes.filter(n=>!n.deleted);
  document.getElementById('stat-total').textContent=active.length;
  document.getElementById('stat-pinned').textContent=active.filter(n=>n.pinned).length;
  document.getElementById('stat-favs').textContent=active.filter(n=>n.favorite).length;
  document.getElementById('stat-private').textContent=active.filter(n=>n.private).length;
  document.getElementById('stat-trash').textContent=trash.length;
  const tw=active.reduce((s,n)=>s+wordCount(n.body),0);
  document.getElementById('stat-words').textContent=tw;
  document.getElementById('stat-avg').textContent=active.length?Math.round(tw/active.length):0;
  const cats=['Study','Personal','Ideas','Work','General'];
  const catColors={Study:'var(--cyan)',Personal:'var(--pink)',Ideas:'var(--accent)',Work:'var(--green)',General:'var(--orange)'};
  const cc={};cats.forEach(c=>cc[c]=active.filter(n=>n.category===c).length);
  const maxC=Math.max(...Object.values(cc),1);
  document.getElementById('catStats').innerHTML=cats.map(c=>`
    <div class="cat-stat"><div class="cat-dot" style="background:${catColors[c]}"></div><span class="cat-stat-name">${c}</span><span class="cat-stat-count">${cc[c]}</span></div>
    <div class="stat-bar"><div class="stat-bar-fill" style="width:${(cc[c]/maxC)*100}%;background:${catColors[c]}"></div></div>`).join('');
  const grid=document.getElementById('activityGrid');
  const ad=Array(28).fill(0);
  active.forEach(n=>{const d=Math.floor((Date.now()-n.updated)/86400000);if(d<28)ad[d]++;});
  const maxA=Math.max(...ad,1);
  grid.innerHTML=ad.reverse().map(v=>{
    let l=0;if(v>0)l=v<maxA*.33?1:v<maxA*.66?2:v<maxA?3:4;
    return`<div class="activity-cell l${l}" title="${v} notes"></div>`;
  }).join('');
}
